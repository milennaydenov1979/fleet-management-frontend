        async function saveVehicle(event) {

            event.preventDefault();

            

            const regNumber = document.getElementById('regNumber').value;

            const brand = document.getElementById('brand').value;


            try {

                const url = currentVehicleId 

                    ? \`\${API_URL}/api/vehicles/\${currentVehicleId}\` 

                    : \`\${API_URL}/api/vehicles\`;

                const method = currentVehicleId ? 'PUT' : 'POST';


                const response = await fetch(url, {

                    method: method,

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify({ regNumber, brand })

                });


                const data = await response.json();


                if (data.success) {

                    showToast(currentVehicleId ? '✅ Обновена!' : '✅ Добавена!');

                    closeVehicleModal();

                }

            } catch (error) {

                showToast('❌ Грешка', 'error');

            }

        }

            }

        }
